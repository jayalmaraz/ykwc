## Coding

### Javascript

Markdown and Mdx parsing is supported via `unified`, and other remark and rehype packages. `next-mdx-remote` allows us to parse `.mdx` and `.md` files in a more flexible manner without touching webpack.

```js
import { LiveReload } from '@remix-run/react';

// Here is a comment!

export default function App() {
  return (
    <html lang="en">
      <head>
        <meta charSet="utf-8" />
        <title>Remix: So great, it's funny!</title>
      </head>
      <body>
        Hello world
        <LiveReload />
      </body>
    </html>
  );
}
```

### Typescript

```ts
type Value = {
  bar: string;
  mee: string;
  value: string;
};

function foo(bar: string): Value {
  let mee = 'Hello, World!';

  // Output variables
  console.log({ mee, bar });

  // Return concat
  return {
    bar,
    mee,
    value: `${mee} ${bar}`,
  };
}

foo('Ligma');
```
